//Notes
Tenants report (tenant profile delta feed).
Returns tenant identity/location fields (tenant code, property/unit ids, name, status, move-in/out, email).
Window: last 3 months based on tenant last-modified or move-in date.
Use: incremental tenant synchronization for contact, occupancy, and downstream legal/arrears correlation.
//End Notes

//Title
LW Portal Export-File07
//End Title

//Select
SELECT
    t.HMYPERSON      AS yardiPersonRowID,
    t.SCODE          AS tenantCode,
    t.HPROPERTY      AS yardiPropertyRowID,
    t.HUNIT          AS yardiUnitRowID,
    t.SFIRSTNAME     AS firstName,
    t.SLASTNAME      AS lastName,
    ts.STATUS        AS status,
    t.DTMOVEIN       AS moveInDate,
    t.DTMOVEOUT      AS moveOutDate,
    t.SEMAIL         AS email
FROM tenant t
LEFT JOIN tenStatus ts ON t.iStatus = ts.iStatus
CROSS JOIN (SELECT CAST(DATEADD(MONTH, -3, GETDATE()) AS date) AS d) c
WHERE
      (t.dtLastModified IS NOT NULL AND CAST(t.dtLastModified AS date) >= c.d)
   OR (t.DTMOVEIN       IS NOT NULL AND CAST(t.DTMOVEIN       AS date) >= c.d)
   OR (t.DTMOVEOUT      IS NOT NULL AND CAST(t.DTMOVEOUT      AS date) >= c.d)
   OR (t.HMYPERSON IN
      (
          SELECT DISTINCT t2.HMYPERSON
          FROM tenant t2
          INNER JOIN tenStatus ts2 ON t2.iStatus = ts2.iStatus
          INNER JOIN unit u2 ON t2.HUNIT = u2.HMY
          INNER JOIN property p2 ON u2.HPROPERTY = p2.HMY
          WHERE ts2.STATUS LIKE 'Current%'
            AND ISNULL(p2.bInactive, 0) = 0
            AND ISNULL(u2.exclude, 0) = 0
            AND (t2.DTMOVEOUT IS NULL OR t2.DTMOVEOUT >= GETDATE())
      ))
//End Select
